\begin{equation*} \begin{array}{l} \text{ From rectangular to cylindrical: } r=\sqrt{x^2+y^2}, \tan(\theta) = y/x \text{ and } z=z;\\ \text{ From cylindrical to rectangular: } x=r\cos(\theta), y=r\sin(\theta) \text{ and } z=z. \end{array} \end{equation*}
