\begin{align*} \frac12(0.2)\amp\Big[(1+0.9608) + (0.9608+0.8521) + (0.8521+0.6977)\\ \amp\quad + ( 0.6977+ 0.5273) +(0.5273 + 0.3679)\Big]\text{.} \end{align*}
