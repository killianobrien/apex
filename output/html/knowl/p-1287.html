\begin{align*} \lim_{x\to\infty}\frac{x^2}{x^2+4} \amp = \lim_{x\to\infty}\frac{x^2/x^2}{x^2/x^2+4/x^2}\\ \amp =\lim_{x\to\infty}\frac{1}{1+4/x^2}\\ \amp =\frac{1}{1+0}\\ \amp = 1\text{.} \end{align*}
