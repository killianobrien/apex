\begin{align*} \lim_{x\to 9} \frac{\sqrt{x}-3}{x-9} \amp =\lim_{x\to 9} \frac{x-9}{(x-9)\left(\sqrt{x}+3\right)}\\ \amp = \lim_{x\to 9 }\frac{1}{\sqrt{x}+3}\\ \amp = \frac{1}{\lim_{x\to 9}\sqrt{x}+\lim_{x \to 9}3}\\ \amp = \frac{1}{\sqrt{\lim_{x\to 9}x}+3}\\ \amp = \frac{1}{\sqrt{3+3}}\\ \amp = \frac{1}{6} \end{align*}
