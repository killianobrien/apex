\begin{equation} \snorm{\text{ proj } _{\,\vec n}\,{\overrightarrow{PQ}}} = \snorm{\frac{\vec n\cdot \overrightarrow{PQ}}{\vnorm n^2}\vec n} = \frac{\abs{\vec n\cdot \overrightarrow{PQ}}}{\vnorm n}\tag{11.6.4} \end{equation}
