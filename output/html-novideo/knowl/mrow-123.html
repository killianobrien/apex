\begin{align*} \lim_{x\to\infty}\frac{x}{\sqrt{x^2+1}} \amp = \lim_{x\to\infty}\frac{x}{\sqrt{x^2+1}}\cdot \frac{\frac{1}{\sqrt{x^2}}}{\frac{1}{\sqrt{x^2}}}\\ \amp = \lim_{x\to\infty} \frac{\frac{x}{\abs{x}}}{\sqrt{\frac{x^2+1}{x^2}}}\\ \amp = \lim_{x\to\infty} \frac{1}{\sqrt{1+\frac{1}{x^2}}} \text{ for } x\gt 0\\ \amp =\frac{1}{\sqrt{1+0}}\\ \amp = 1 \end{align*}
