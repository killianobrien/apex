\begin{align*} \int_1^\infty \frac{\ln(x) }{x^2}\, dx \amp = \lim_{b\to\infty} \int_1^b \frac{\ln(x) }{x^2}\, dx\\ \amp = \lim_{b\to\infty} -\frac1x\ln(x) \Big|_1^b + \int_1^b\frac1{x^2}\, dx\\ \amp = \lim_{b\to\infty} -\frac1x\ln(x) -\frac 1x\Big|_1^b\\ \amp = \lim_{b\to\infty}1-\frac1b-\frac{\ln(b) }{b}. \text{ Apply L'Hospital's Rule: }\\ \amp = 1 \end{align*}
