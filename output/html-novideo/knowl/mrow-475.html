\begin{align*} -50+\frac{130x}{\sqrt{x^2+1000^2}} \amp = 0\\ \frac{130x}{\sqrt{x^2+1000^2}} \amp = 50\\ \frac{130^2x^2}{x^2+1000^2} \amp = 50^2\\ 130^2x^2 \amp = 50^2(x^2+1000^2)\\ 130^2x^2-50^2x^2 \amp = 50^2\cdot1000^2\\ (130^2-50^2)x^2 \amp = 50,000^2\\ x^2 \amp = \frac{50,000^2}{130^2-50^2}\\ x \amp = \frac{50,000}{\sqrt{130^2-50^2}}\\ x \amp = \frac{50,000}{120} =\frac{1250}3\approx 416.67 \end{align*}
