\begin{align*} \inflim \left(a_n\right)^{1/n} \amp = \inflim \left(\left(\frac{3n+1}{5n-2}\right)^n\right)^{1/n}\\ \amp =\inflim \frac{3n+1}{5n-2} = \frac 35\text{.} \end{align*}
