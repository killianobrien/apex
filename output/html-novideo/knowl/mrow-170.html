\begin{align*} \fp(x) \amp = \lim_{h\to 0} \frac{\sin(x+h)-\sin(x)}{h} \amp \amp \text{Derivative definition}\\ \amp = \lim_{h\to 0} \frac{\sin(x)\cos(h)+\cos(x)\sin(h)-\sin(x)}{h} \amp \amp \text{Angle addition identity}\\ \amp = \lim_{h\to 0} \frac{\sin(x)(\cos(h)-1) + \cos(x)\sin(h)}{h} \amp \amp \text{Regrouped and factored}\\ \amp = \lim_{h\to 0} \left(\frac{\sin(x)(\cos(h)-1)}{h} + \frac{\cos(x)\sin(h)}{h} \right) \amp \amp \text{Split into two fractions}\\ \amp = \lim_{h\to 0} \sin(x) \cdot \lim_{h\to 0}\frac{\cos(h)-1}{h}\\ \amp\quad{}+\lim_{h\to 0}\cos(x)\cdot \lim_{h\to 0}\frac{\sin(h)}{h} \amp \amp \text{Product/sum limit rules}\\ \amp = \sin(x)\cdot 0 + \cos(x) \cdot 1 \amp \amp \text{Applied } \knowl{./knowl/thm_special_limits.html}{\text{Theorem 1.3.12}}\\ \amp = \cos(x). \amp \amp \text{(Are you surprised?)} \end{align*}
